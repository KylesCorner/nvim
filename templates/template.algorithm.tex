% Macro to draw a vertical stack from comma-separated values
\newcommand{\drawstack}[1]{%
  \begin{SafeTikz}
    \foreach \elem [count=\i] in {#1} {
      \node[draw, minimum width=2cm, minimum height=1cm, anchor=south] (n\i) at (0, \i - 1) {\elem};
    }
  \end{SafeTikz}
}
